from __future__ import annotations

from datetime import date, datetime
from pathlib import Path

from reportlab.lib.colors import HexColor, white
from reportlab.lib.enums import TA_RIGHT
from reportlab.lib.pagesizes import A4
from reportlab.lib.styles import ParagraphStyle
from reportlab.lib.utils import ImageReader
from reportlab.pdfbase import pdfmetrics
from reportlab.pdfbase.ttfonts import TTFont
from reportlab.pdfgen.canvas import Canvas
from reportlab.platypus import Paragraph

ROOT = Path(__file__).resolve().parents[1]
NAVY, RED, OFFWHITE = HexColor("#102A43"), HexColor("#E50012"), HexColor("#FBFAF8")
GRAY, LIGHTBLUE, BLACK = HexColor("#777777"), HexColor("#9FB3C8"), HexColor("#181818")
LINE = HexColor("#D8DEE5")


def register_fonts() -> None:
    font_dir = ROOT / "assets/fonts"
    for name, filename in [("Pretendard", "Pretendard-Regular.ttf"), ("Pretendard-SemiBold", "Pretendard-SemiBold.ttf"), ("Pretendard-Bold", "Pretendard-Bold.ttf"), ("Pretendard-ExtraBold", "Pretendard-ExtraBold.ttf")]:
        pdfmetrics.registerFont(TTFont(name, str(font_dir / filename)))


def paragraph(canvas: Canvas, text: str, x: float, y_top: float, width: float, height: float, style: ParagraphStyle) -> float:
    """주어진 영역을 넘지 않도록 말줄임표를 적용해 문단을 그립니다."""
    safe = text.replace("&", "&amp;")
    p = Paragraph(safe, style)
    _, h = p.wrap(width, height)
    if h > height:
        low, high, best = 1, len(text), ""
        while low <= high:
            mid = (low + high) // 2
            candidate = text[:mid].rstrip() + "…"
            test = Paragraph(candidate.replace("&", "&amp;"), style)
            _, test_h = test.wrap(width, height)
            if test_h <= height:
                best, low = candidate, mid + 1
            else:
                high = mid - 1
        p = Paragraph((best or "…").replace("&", "&amp;"), style)
        _, h = p.wrap(width, height)
    p.drawOn(canvas, x, y_top - h)
    return h


def build_pdf(output: Path, articles: list, report_date: date, cfg: dict) -> None:
    register_fonts()
    output.parent.mkdir(parents=True, exist_ok=True)
    cards_per_page = cfg.get("cards_per_page", 5)
    section_order = list(cfg["sections"])
    pages: list[tuple[str, list]] = []
    for section in section_order:
        rows = [a for a in articles if a.section == section]
        if not rows:
            continue
        for start in range(0, len(rows), cards_per_page):
            pages.append((section, rows[start:start + cards_per_page]))

    w, h = A4
    c = Canvas(str(output), pagesize=A4, pageCompression=1)
    c.setTitle(f"{cfg['briefing_title']} {report_date:%Y.%m.%d}")
    total = len(pages) + 1

    # 1페이지: 섹션별 대표 기사 한 건을 모은 오늘의 헤드라인
    c.setFillColor(white); c.rect(0, 0, w, h, stroke=0, fill=1)
    c.setFillColor(NAVY); c.rect(0, h - 96, w, 96, stroke=0, fill=1)
    c.setFillColor(LIGHTBLUE); c.setFont("Pretendard-SemiBold", 10); c.drawString(34, h - 43, "N E W S   C L I P P I N G")
    c.setFillColor(white); c.setFont("Pretendard-ExtraBold", 22); c.drawString(34, h - 70, cfg["briefing_title"])
    c.setFont("Pretendard-Bold", 14); c.drawRightString(w - 34, h - 52, f"{report_date:%Y.%m.%d}")
    c.setFillColor(LIGHTBLUE); c.setFont("Pretendard-SemiBold", 9); c.drawRightString(w - 34, h - 69, f"1 / {total}")
    c.setStrokeColor(white); c.setLineWidth(1.5); c.line(34, h - 87, w - 34, h - 87)
    c.setLineWidth(0.5); c.line(34, h - 92, w - 34, h - 92)
    c.setFillColor(RED); c.rect(34, h - 132, 7, 26, stroke=0, fill=1)
    c.setFillColor(NAVY); c.setFont("Pretendard-ExtraBold", 24); c.drawString(50, h - 128, "오늘의 헤드라인")
    c.setStrokeColor(NAVY); c.setLineWidth(1.2); c.line(34, h - 143, w - 34, h - 143)

    headline_groups = []
    for section in section_order:
        section_articles = [a for a in articles if a.section == section]
        if section_articles:
            headline_groups.append((section, section_articles[:3]))
    headline_top, headline_bottom = h - 151, 76
    group_h = (headline_top - headline_bottom) / max(1, len(headline_groups))
    headline_style = ParagraphStyle("headline", fontName="Pretendard-SemiBold", fontSize=11.5, leading=13, textColor=BLACK)
    for idx, (section, group_articles) in enumerate(headline_groups):
        group_top = headline_top - idx * group_h
        c.setFillColor(RED); c.setFont("Pretendard-Bold", 13.5); c.drawString(34, group_top - 14, section)
        c.setStrokeColor(NAVY); c.setLineWidth(0.9); c.line(34, group_top - 20, w - 34, group_top - 20)
        for article_idx, a in enumerate(group_articles):
            article_top = group_top - 28 - article_idx * 19
            try:
                article_date = datetime.fromisoformat(a.published_at).strftime("%y.%m.%d")
            except (TypeError, ValueError):
                article_date = report_date.strftime("%y.%m.%d")
            c.setFillColor(HexColor("#8A939D")); c.setFont("Pretendard-SemiBold", 9.5)
            c.drawRightString(w - 34, article_top - 9, f"{a.publisher} | {article_date}")
            title_h = paragraph(c, a.title, 34, article_top, w - 205, 13.5, headline_style)
            c.linkURL(a.url, (34, article_top - title_h - 1, w - 195, article_top + 2), relative=0, thickness=0)
            c.setStrokeColor(LINE); c.setLineWidth(0.45); c.line(34, article_top - 16, w - 34, article_top - 16)
    c.setFillColor(NAVY); c.setFont("Pretendard-SemiBold", 10.5); c.drawString(34, 42, "※ 기사 제목 클릭 시 원문 페이지로 이동합니다.")
    logo = ROOT / "assets/dts3.jpg"
    if logo.exists():
        c.drawImage(ImageReader(str(logo)), w - 204, 29, width=170, height=28, preserveAspectRatio=True)
    c.showPage()

    for page_no, (section, rows) in enumerate(pages, 1):
        c.setFillColor(white); c.rect(0, 0, w, h, stroke=0, fill=1)
        c.setFillColor(NAVY); c.rect(0, h - 96, w, 96, stroke=0, fill=1)
        c.setFillColor(LIGHTBLUE); c.setFont("Pretendard-SemiBold", 10); c.drawString(34, h - 43, "N E W S   C L I P P I N G")
        c.setFillColor(white); c.setFont("Pretendard-ExtraBold", 22); c.drawString(34, h - 70, cfg["briefing_title"])
        c.setFont("Pretendard-Bold", 14); c.drawRightString(w - 34, h - 52, f"{report_date:%Y.%m.%d}")
        c.setFillColor(LIGHTBLUE); c.setFont("Pretendard-SemiBold", 9); c.drawRightString(w - 34, h - 69, f"{page_no + 1} / {total}")
        c.setStrokeColor(white); c.setLineWidth(1.5); c.line(34, h - 87, w - 34, h - 87)
        c.setLineWidth(0.5); c.line(34, h - 92, w - 34, h - 92)

        continued = rows[0] is not [a for a in articles if a.section == section][0] if rows else False
        section_label = f"{section}  (계속)" if continued else section
        c.setFillColor(RED); c.rect(34, h - 130, 6, 23, stroke=0, fill=1)
        c.setFillColor(NAVY); c.setFont("Pretendard-Bold", 18); c.drawString(49, h - 126, section_label)
        c.setStrokeColor(NAVY); c.setLineWidth(1.2); c.line(34, h - 139, w - 34, h - 139)

        top, bottom = h - 146, 75
        row_h = (top - bottom) / cards_per_page
        title_style = ParagraphStyle("title", fontName="Pretendard-Bold", fontSize=20, leading=24, textColor=BLACK)
        body_style = ParagraphStyle("body", fontName="Pretendard", fontSize=18, leading=22, textColor=GRAY)
        for idx, a in enumerate(rows):
            row_top = top - idx * row_h
            try:
                article_date = datetime.fromisoformat(a.published_at).strftime("%y.%m.%d")
            except (TypeError, ValueError):
                article_date = report_date.strftime("%y.%m.%d")
            title_top = row_top - 10
            title_h = paragraph(c, a.title, 34, title_top, w - 68, 24, title_style)
            c.linkURL(a.url, (34, title_top - title_h - 3, w - 34, title_top + 4), relative=0, thickness=0)
            summary_top = title_top - title_h - 3
            paragraph(c, a.summary, 34, summary_top, w - 68, 44, body_style)
            c.setFillColor(HexColor("#8A939D")); c.setFont("Pretendard-SemiBold", 12.6)
            c.drawRightString(w - 34, row_top - row_h + 13, f"{a.publisher} | {article_date}")
            c.setStrokeColor(LINE); c.setLineWidth(0.7); c.line(34, row_top - row_h, w - 34, row_top - row_h)

        c.setFillColor(NAVY); c.setFont("Pretendard-SemiBold", 10.5)
        c.drawString(34, 42, "※ 기사 제목 클릭 시 원문 페이지로 이동합니다.")
        logo = ROOT / "assets/dts3.jpg"
        if logo.exists():
            c.drawImage(ImageReader(str(logo)), w - 204, 29, width=170, height=28, preserveAspectRatio=True)
        c.showPage()
    c.save()
